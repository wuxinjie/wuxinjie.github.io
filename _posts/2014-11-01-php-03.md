* 求两个日期的差数，例如 2007-2-5 ~ 2007-3-6  的日期差数

	function get_days($date1,$date2)
	{ 
		$time1=strtotime($date1);
		$time2=strtotime($date2);
		return abs($time1-$time2)/86400;
